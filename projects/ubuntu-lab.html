2. Download Ubuntu Server ISO

wget https://releases.ubuntu.com/22.04/ubuntu-22.04.3-live-server-amd64.iso
sha256sum ubuntu-22.04.3-live-server-amd64.iso

3. Network Configuration (Static IP)

network:
  version: 2
  ethernets:
    ens33:
      addresses:
        - 192.168.56.110/24
      gateway4: 192.168.56.1
      nameservers:
        addresses:
          - 8.8.8.8
          - 8.8.4.4

4. Post-Installation Setup

sudo apt update && sudo apt upgrade -y
sudo apt install -y vim curl wget git htop net-tools
sudo timedatectl set-timezone Asia/Jakarta

5. Firewall & Security Hardening

sudo ufw enable
sudo ufw allow 22/tcp
sudo ufw status verbose
    
PermitRootLogin no
MaxAuthTries 3
ClientAliveInterval 300
ClientAliveCountMax 2

6. Fail2Ban Protection

sudo apt install fail2ban -y

[sshd]
enabled = true
maxretry = 3
bantime = 3600

7. Verification Checklist

hostnamectl
ip addr
systemctl status sshd
systemctl status ufw
